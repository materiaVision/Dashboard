${current.temp_C}°C
${current.weatherDesc[0].value}
`; // DNI const daysPL = ["Nd","Pon","Wt","Śr","Czw","Pt","Sob"]; // FORECAST (NAPRAWIONY) let html = ""; for(let i=0;i<5;i++){ const day = data.weather[i]; const date = new Date(day.date); const dayName = daysPL[date.getDay()]; // BEZPIECZNA IKONA let icon = day.hourly && day.hourly[0] ? day.hourly[0].weatherIconUrl[0].value : current.weatherIconUrl[0].value; html += `
${dayName}
${day.avgtempC}°
`; } document.getElementById("forecast").innerHTML = html; }catch(e){ console.log("forecast error", e); document.getElementById("forecast").innerHTML = "Brak danych"; } }
